## sel4-test dev仓库设置

在sel4-test文件夹下创建一个shell脚本`set_seL4-test_dev.sh`

``` shell
#this shell should run in the home directory 
echo "start set and update sel4-test dev repo..."

if [ ! -d "projects" ];then
	echo "please run at home directory! shell failed!"
	exit
else
	printf "now at path: "
	pwd
fi

echo "start set and update kernel as sel4-LA..."
if [ -d "kernel/.git" ];then
	echo "repository sel-LA had been set"
else 
	mv kernel kernel_old
	git clone -b dev git@github.com:tyyteam/seL4-LA.git kernel
	if [ $? -eq 0 ]; then
		echo "git clone sel4-LA finished"
	else
		echo "git clone sel4-LA failed!"
		exit
	fi
fi
cd kernel
echo "pull from seL4-LA"
git pull
cd ..

echo "start set and update seL4 from sel4-tools..."
cd tools
if [ -d "seL4/.git" ];then
	echo "repository sel4-tools had been set"
else 
	mv seL4 seL4_old
	git clone -b dev git@github.com:tyyteam/seL4_tools.git seL4
	if [ $? -eq 0 ]; then
		echo "git clone sel4-tools finished"
	else
		echo "git clone sel4-tools failed!"
		exit
	fi
fi
cd seL4
echo "pull from sel4-tools"
git pull
cd ..

cd ../projects
echo "start set and update musllibc..."
if [ -d "musllibc/.git" ];then
	echo "repository musllibc had been set"

else 
	mv musllibc musllibc_old
	git clone -b dev git@github.com:tyyteam/musllibc.git
	if [ $? -eq 0 ]; then
		echo "git clone musllibc finished"
	else
		echo "git clone musllibc failed!"
		exit
	fi
fi
cd musllibc
echo "pull from musllibc"
git pull
cd ..

echo "start set and update sel4runtime..."
if [ -d "sel4runtime/.git" ];then
	echo "repository selruntime had been set"

else 
	mv sel4runtime sel4runtime_old
	git clone -b dev git@github.com:tyyteam/sel4runtime.git
	if [ $? -eq 0 ]; then
		echo "git clone sel4runtime finished"
	else
		echo "git clone sel4runtime failed!"
		exit
	fi
fi
cd sel4runtime
echo "pull from sel4runtime"
git pull
cd ..

echo "start set and update sel4test..."
if [ -d "sel4test/.git" ];then
	echo "repository sel4test had been set"

else 
	mv sel4test sel4test_old
	git clone -b dev git@github.com:tyyteam/sel4test.git
	if [ $? -eq 0 ]; then
		echo "git clone sel4test finished"
	else
		echo "git clone sel4test failed!"
		exit
	fi
fi
cd sel4test
echo "pull from sel4test"
git pull
cd ..

cd ..
echo "sel4-test dev repo has been updated"
```

chmod u+x set_seL4-test_dev.sh

过段时间运行一下这个脚本来更新。
